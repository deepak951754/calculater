    <script>
        const screen = document.getElementById('screen');
        let resetScreen = false;

        // Appends values onto the calculation string stream
        function appendValue(value) {
            // Guard conditions to manage starting state or refresh targets
            if (screen.innerText === '0' || screen.innerText === 'Error' || resetScreen) {
                screen.innerText = '';
                resetScreen = false;
            }
            screen.innerText += value;
        }

        // Destructive utility to wipe calculations clean
        function clearScreen() {
            screen.innerText = '0';
        }

        // Safely processes string assessment evaluation blocks
        function calculateResult() {
            try {
                let expression = screen.innerText;
                
                // If input is empty or invalid edge fallback
                if (!expression) return;

                // Evaluate the basic math string stream expression safely 
                let result = Function('"use strict";return (' + expression + ')')();

                // Format decimals safely if values evaluate long
                if (result % 1 !== 0) {
                    result = parseFloat(result.toFixed(4));
                }

                screen.innerText = result;
                resetScreen = true; // Flips state to overwrite screen on next key-press
            } catch (error) {
                screen.innerText = 'Error';
                resetScreen = true;
            }
        }
    </script>
